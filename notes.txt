.gitignore // carpeta para ignorar

git init // iniciar repositorio git
ls // ver el contenido de una carpeta
git status // revisar a que le hago segumiento
git commit // hacer commit
git commit -m "<Mensaje>" // hacer commit con mensaje si entrar a bin
git log // info
git checkout -- <file> // el archivo vuelve a su ultimo commit
git diff <file> // me muestra la diferencia entre el mi archivo actual y el ultimo commit
git add <file> // actualizar datos del commit
git add . // actualizar todos los datos del commit
git restore <file> // el archivo vuelve a su ultimo commit
:wq // terminar commit
git branch // manejo de versiones alternativas
git branch <name> // crea una nueva version alternativa
git checkout <name> // selecciona la version alternativa a trabajar
// se usa para hacer una subdivision del codigo por partes como por ejemplo
